1️⃣ Which products are performing above average? (Price > Average of all products)

select Product.ProductName ,Product.Price 
from Product
where Price > (select AVG(Price) from Product);
with AvgPrice as (
  select AVG(Price) as Avg_Price
  from Product)
select Product.ProductName ,Product.Price
from Product cross join AvgPrice
where Price > Avg_Price;
create view AvgPrice as 
select Product.ProductName ,Product.Price
from Product
where Price > (select AVG(Price) from Product);

select * from AvgPrice;
select ProductName , avg_price
from (
  select Product.ProductName, Product.Price,
         AVG(Price) over(partition by categoryid) as avg_price
  from Product) AvgPrice
where Price > avg_price;

2️⃣ Find products above average compared to their category competitors

select p.ProductName , p.Price , p.CategoryID
from Product p 
join (select CategoryID , AVG(price) as avg_price
      from Product group by CategoryID) AvgPrice
on p.CategoryID = AvgPrice.CategoryID
where Price > avg_price;
with AvgPrice as (
  select CategoryID, AVG(price) as avg_price
  from Product group by CategoryID)
select p.ProductName, p.Price, p.CategoryID
from Product p cross join AvgPrice
where Price > avg_price;
create view Av_Price as 
select p.ProductName , p.Price , p.CategoryID
from Product p 
join (select CategoryID , AVG(price) as avg_price
      from Product group by CategoryID) A_Price
on p.CategoryID = A_Price.CategoryID
where price > avg_price;

select * from Av_Price;
select ProductName , Price , CategoryID, avg_price
from (
  select p.ProductName , p.Price , p.CategoryID,
         avg(price) over(partition by categoryid) as avg_price
  from Product p) AvgPrice
where Price > avg_price;

3️⃣ Identify champion products (highest price per category)

select P.ProductName , P.CategoryID , Max_price
from Product P
join (select CategoryID, max(Price) as Max_price
      from Product group by CategoryID) MaxPrice
on p.CategoryID = MaxPrice.CategoryID
where Price = Max_price
order by Max_price desc;
with MaxPrice as (
  select CategoryID, max(Price) as Max_price
  from Product group by CategoryID)
select P.ProductName ,P.CategoryID, P.Price
from Product P cross join MaxPrice
where Price = Max_price
order by Max_price desc;
create view MaxPrice as 
select p.* , Max_Price
from Product p
join (select CategoryID, max(Price) as Max_Price
      from Product group by CategoryID) Max_P
on p.CategoryID = Max_P.CategoryID
where Price = Max_Price;

select * from MaxPrice;
